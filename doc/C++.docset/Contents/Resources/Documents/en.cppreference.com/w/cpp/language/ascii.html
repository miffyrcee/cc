Note: in Unicode, the ASCII character block is known as U+0000..U+007F Basic Latin.

[edit] Example

Run this code
#include <iostream>
int main()
{
    std::cout << "Printable ASCII [32..126]:\n";
    for (char c{' '}; c <= '~'; c++) {
        std::cout << c << ((c+1) % 32 ? ' ' : '\n');
    }
}
